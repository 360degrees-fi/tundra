  9 
 10 /**
 11  * @namespace main namespace.
 12  */
 13 var realXtend = realXtend || {};
 14 
 15 /**
 16  * @namespace PathFollow namespace contains functionality to follow paths. 
 17  */
 18 realXtend.PathFollow = realXtend.PathFollow || {};
 19 
 20 
 21 /**
 22  * Calculates next waypoint and direction of the path.
 23  * @param pathid Id of the Path.
 24  * @param actualpointid Id of the previous point.
 25  * @param direction Direction of the Path.
 26  * @returns {Array} [NextPoint,Direction].
 27  * @static
 28  */
 29 realXtend.PathFollow.getNextWaypoint = function(pathid,actualpointid,direction)
 30 {	
 31 	var path=scene.GetEntityRaw(pathid);
 32 	var dynamiccomponent=path.GetComponentRaw("EC_DynamicComponent");
 33 	var list=dynamiccomponent.GetAttribute("WayPointList");
 34 	var elem = list.split(',');	
 35 	var pointnum = elem.length;	
 36 	var random=dynamiccomponent.GetAttribute("Random");
 37 	var type=dynamiccomponent.GetAttribute("Type");	
 38 	var elemi;
 39 	if(actualpointid!=0){
 40 		var actualpoint=scene.GetEntityRaw(actualpointid);
 41 		if(random==false){
 42 			if(type==1){
 43 				for (i=0;i<pointnum;i++){
 44 					var where=scene.GetEntityByNameRaw("WayPoint["+elem[i]+"]");
 45 					if(typeof(where) != "undefined" && where.id==actualpointid){
 46 						if(direction==1){
 47 							elemi=scene.GetEntityByNameRaw("WayPoint["+elem[pointnum-1]+"]");
 48 							if(typeof(elemi) != "undefined" && where.id==elemi.id){
 49 								return [0,direction];
 50 							}
 51 							else if(typeof(elemi) != "undefined" && where.id!=elemi.id){
 52 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i+1]+"]");
 53 								return [nextpoint.id,direction];
 54 							}
 55 						}
 56 						else if(direction==-1){
 57 							elemi=scene.GetEntityByNameRaw("WayPoint["+elem[0]+"]");
 58 							if(typeof(elemi) != "undefined" && where.id==elemi.id){
 59 								return [0,direction];
 60 							}
 61 							else if(typeof(elemi) != "undefined" && where.id!=elemi.id){
 62 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i-1]+"]");
 63 								return [nextpoint.id,direction];
 64 							}
 65 						}
 66 					}
 67 				}
 68 			}
 69 			else if(type==2){
 70 				for (i=0;i<pointnum;i++){
 71 					var where=scene.GetEntityByNameRaw("WayPoint["+elem[i]+"]");
 72 					if(typeof(where) != "undefined" && where.id==actualpointid){
 73 						if(direction==1){
 74 							elemi=scene.GetEntityByNameRaw("WayPoint["+elem[pointnum-1]+"]");
 75 							if(typeof(elemi) != "undefined" && where.id==elemi.id){
 76 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i-1]+"]");
 77 								direction=-1;
 78 								return [nextpoint.id,direction];
 79 							}
 80 							else if(typeof(elemi) != "undefined" && where.id!=elemi.id){
 81 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i+1]+"]");
 82 								return [nextpoint.id,direction];
 83 							}
 84 						}
 85 						else if(direction==-1){
 86 							elemi=scene.GetEntityByNameRaw("WayPoint["+elem[0]+"]");
 87 							if(typeof(elemi) != "undefined" && where.id==elemi.id){
 88 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[1]+"]");
 89 								direction=1;
 90 								return [nextpoint.id,direction];
 91 							}
 92 							else if(typeof(elemi) != "undefined" && where.id!=elemi.id){
 93 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i-1]+"]");
 94 								return [nextpoint.id,direction];
 95 							}
 96 						}
 97 					}
 98 				}
 99 			}
100 			else if(type==3){
101 				for (i=0;i<pointnum;i++){
102 					var where=scene.GetEntityByNameRaw("WayPoint["+elem[i]+"]");
103 					if(typeof(where) != "undefined" && where.id==actualpointid){
104 						if(direction==1){
105 							elemi=scene.GetEntityByNameRaw("WayPoint["+elem[pointnum-1]+"]");
106 							if(typeof(elemi) != "undefined" && where.id==elemi.id){
107 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[0]+"]");
108 								return [nextpoint.id,direction];
109 							}
110 							else if(typeof(elemi) != "undefined" && where.id!=elemi.id){
111 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i+1]+"]");
112 								return [nextpoint.id,direction];
113 							}
114 						}
115 						else if(direction==-1){
116 							elemi=scene.GetEntityByNameRaw("WayPoint["+elem[0]+"]");
117 							if(typeof(elemi) != "undefined" && where.id==elemi.id){
118 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[pointnum-1]+"]");
119 								return [nextpoint.id,direction];
120 							}
121 							else if(typeof(elemi) != "undefined" && where.id!=elemi.id){
122 								var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i-1]+"]");
123 								return [nextpoint.id,direction];
124 							}
125 						}
126 					}
127 				}
128 			}
129 		}
130 		else if(random==true){
131 			var i = Math.floor(Math.random() * pointnum);
132 			var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[i]+"]");
133 				return [nextpoint.id,direction];
134 		}
135 	}
136 	else if(actualpointid==0){
137 		if(type==1 || type ==2){
138 			if(direction==1){
139 				var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[0]+"]");
140 				return [nextpoint.id,direction];
141 			}
142 			else if(direction==-1){
143 				var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[pointnum-1]+"]");
144 				return [nextpoint.id,direction];
145 			}
146 		}
147 		if(type==3){
148 			var nextpoint=scene.GetEntityByNameRaw("WayPoint["+elem[0]+"]");
149 			return [nextpoint.id,direction];
150 		}
151 	}
152 };
