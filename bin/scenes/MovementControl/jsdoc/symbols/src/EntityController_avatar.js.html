 24 realXtend.EntityController.SimpleAvatar = function()
 25 {
 26 	this.motion = new Vector3df();
 27 	this.rotate = 0;
 28 	this.mouse_rotate_sensitivity = 0.3;
 29 	this.avatar_mass = 10;
 30 	//this.moveVec = new Vector3df();
 31 	this.rotationVec = new Vector3df();
 32 	this.rotationVec.z = 1;
 33 	
 34 	
 35 	// Needed bools for logic
 36 	this.isserver = server.IsRunning();
 37 	this.own_avatar = false;
 38 	this.fish_created = false;
 39 	
 40 	// Gestures
 41 	this.listenGesture = false;
 42 	
 43 	//Create Entity Controller
 44 //	engine.IncludeFile("local://RigidBodyController.js");
 45 	this.avatar_rigid_body = new realXtend.EntityController.RigidBodyController();  
 46 	
 47 	//Create Animation sync
 48 	this.avatar_animation_sync = new realXtend.EntityController.AvatarAnimationSync();
 49 	this.avatar_rigid_body.AddSyncObject(this.avatar_animation_sync);
 50 	
 51 	// Create avatar on server, and camera & inputmapper on client
 52 	if (this.isserver) {
 53 	    this.ServerInitialize();
 54 	} else {
 55 	    this.ClientInitialize();
 56 	}
 57 
 58 };

 62 realXtend.EntityController.SimpleAvatar.prototype.ServerInitialize = function () {	
 63     var avatar = me.GetOrCreateComponentRaw("EC_Avatar");
 64     var voice = me.GetOrCreateComponentRaw("EC_TtsVoice");
 65        
 66     var rigidbody = me.rigidbody; 
 67 
 68     // Set the avatar appearance. This creates the mesh & animationcontroller, once the avatar asset has loaded
 69     // Note: for now, you need the default_avatar.xml in your bin/data/assets folder
 70     // Set the avatar appearance. This creates the mesh & animationcontroller, once the avatar asset has loaded
 71     var r = avatar.appearanceRef;
 72     r.ref = "local://default_avatar.xml";
 73     avatar.appearanceRef = r;
 74 
 75     // Set voice to avatar
 76     voice.voice = "Spanish male";
 77 
 78     // Set physics properties
 79     var sizeVec = new Vector3df();
 80     sizeVec.z = 2.4;
 81     sizeVec.x = 0.5;
 82     sizeVec.y = 0.5;
 83     rigidbody.mass = this.avatar_mass;
 84     rigidbody.shapeType = 3; // Capsule
 85     rigidbody.size = sizeVec;
 86     rigidbody.friction = 0.3;
 87 
 88     var angularVec = new Vector3df();
 89     angularVec.z = 1;
 90     rigidbody.angularFactor = angularVec; // Set zero angular factor so that body stays upright
 91     
 92     var avatar_properties = me.dynamiccomponent;
 93     if (avatar_properties)
 94     {
 95     	avatar_properties.SetAttribute("rotationMaskX",this.rotationVec.x);
 96     	avatar_properties.SetAttribute("rotationMaskY",this.rotationVec.y);
 97     }    
 98 };

104 realXtend.EntityController.SimpleAvatar.prototype.ClientInitialize = function() {
105     // Check if this is our own avatar
106     // Note: bad security. For now there's no checking who is allowed to invoke actions
107     // on an entity, and we could theoretically control anyone's avatar
108     if (me.GetName() == "Avatar" + client.GetConnectionID()) {
109         this.own_avatar = true;
110         this.ClientCreateInputMapper();
111 
112         //Create Camera sync
113         this.client_avatar_camera_sync = new realXtend.EntityController.AvatarCameraSync(7.0, 1.0);
114         this.avatar_rigid_body.AddSyncObject(this.client_avatar_camera_sync);
115         
116         //connect actions
117         me.Action("KeyMove").Triggered.connect(this, this.ClientKeyMove);
118         me.Action("KeyRotate").Triggered.connect(this, this.ClientKeyRotate);
119         me.Action("KeyStop").Triggered.connect(this, this.ClientKeyStop);
120         me.Action("KeyStopRotate").Triggered.connect(this, this.ClientKeyStopRotate);
121         
122         me.Action("MouseLookX").Triggered.connect(this, this.ClientHandleLookX);        
123         me.Action("CheckState").Triggered.connect(this, this.ClientHandleCheckState);
124         var soundlistener = me.GetOrCreateComponentRaw("EC_SoundListener");
125         soundlistener.active = true;
126 
127     }
128     else
129     {
130         // Make hovering name tag for other clients
131         var clientName = me.GetComponentRaw("EC_Name");
132         if (clientName != null) {
133             // Description holds the actual login name
134             if (clientName.description != "") {
135                 var hoveringWidget = me.GetOrCreateComponentRaw("EC_HoveringWidget", 2, false);
136                 if (hoveringWidget != null) {
137                     hoveringWidget.SetNetworkSyncEnabled(false);
138                     hoveringWidget.SetTemporary(true);
139                     hoveringWidget.InitializeBillboards();
140                     hoveringWidget.SetButtonsDisabled(true);
141                     hoveringWidget.SetText(clientName.description);
142                     hoveringWidget.SetFontSize(100);
143                     hoveringWidget.SetTextHeight(200);
144                     hoveringWidget.Show();
145                 }
146             }
147         }
148     }
149 };

151 realXtend.EntityController.SimpleAvatar.prototype.ClientHandleLookX = function(param)
152 {
153     if (this.client_avatar_camera_sync)
154     {
155     	if (this.client_avatar_camera_sync.IsTripod())
156     		this.client_avatar_camera_sync.RotateZ(param);
157     	else
158     	{
159     		var move=parseInt(param);
160     		
161     		move = -move * this.mouse_rotate_sensitivity;    		
162     		//exec rotate action on client
163     		me.Exec(1, "Rotate", ["0","0",move.toString(),"true"]);	    		
164     	}
165     }
166     	
167 };

170 realXtend.EntityController.SimpleAvatar.prototype.ClientHandleCheckState = function()
171 {
172 	var inputmapper = me.inputmapper;
173 	if ((this.client_avatar_camera_sync != null) && (inputmapper != null)) {

175         if (inputmapper.enabled != active)
176             inputmapper.enabled = active;
177     }

180 realXtend.EntityController.SimpleAvatar.prototype.ClientCreateInputMapper = function() {
181     // Create a nonsynced inputmapper
182     var inputmapper = me.GetOrCreateComponentRaw("EC_InputMapper", 2, false);
183     inputmapper.contextPriority = 101;
184     inputmapper.takeMouseEventsOverQt = false;
185     inputmapper.takeKeyboardEventsOverQt = false;
186     inputmapper.modifiersEnabled = false;
187     inputmapper.keyrepeatTrigger = false; // Disable repeat keyevent sending over network, not needed and will flood network
188     inputmapper.executionType = 1; // Execute actions on client
189     inputmapper.RegisterMapping("W", "KeyMove(forward)", 1);
190     inputmapper.RegisterMapping("S", "KeyMove(back)", 1);
191     inputmapper.RegisterMapping("A", "KeyMove(left)", 1);
192     inputmapper.RegisterMapping("D", "KeyMove(right))", 1);
193     inputmapper.RegisterMapping("Up", "KeyMove(forward)", 1);
194     inputmapper.RegisterMapping("Down", "KeyMove(back)", 1);
195     inputmapper.RegisterMapping("Left", "KeyRotate(left)", 1);
196     inputmapper.RegisterMapping("Right", "KeyRotate(right))", 1);
197     inputmapper.RegisterMapping("F", "ToggleFly()", 1);
198     inputmapper.RegisterMapping("Q", "Gesture(wave)", 1);
199     inputmapper.RegisterMapping("Space", "KeyMove(up)", 1);
200     inputmapper.RegisterMapping("C", "KeyMove(down)", 1);
201     inputmapper.RegisterMapping("W", "KeyStop(forward)", 3);
202     inputmapper.RegisterMapping("S", "KeyStop(back)", 3);
203     inputmapper.RegisterMapping("A", "KeyStop(left)", 3);
204     inputmapper.RegisterMapping("D", "KeyStop(right)", 3);
205     inputmapper.RegisterMapping("Up", "KeyStop(forward)", 3);
206     inputmapper.RegisterMapping("Down", "KeyStop(back)", 3);
207     inputmapper.RegisterMapping("Left", "KeyStopRotate(left)", 3);
208     inputmapper.RegisterMapping("Right", "KeyStopRotate(right))", 3);
209     inputmapper.RegisterMapping("Space", "KeyStop(up)", 3);
210     inputmapper.RegisterMapping("C", "KeyStop(down)", 3);
211     inputmapper.RegisterMapping("R", "ToggleRun()", 1);
212     
213 
214     // Connect gestures
215     var inputContext = inputmapper.GetInputContext();
216     inputContext.GestureStarted.connect(this, this.GestureStarted);
217     inputContext.GestureUpdated.connect(this, this.GestureUpdated);          
218     
219     // mouse click to raycast point
220     inputContext.MouseLeftPressed.connect(this, this.ClientHandleMouseLeftPressed);
221 };

223 realXtend.EntityController.SimpleAvatar.prototype.ClientHandleMouseLeftPressed = function(mouseevent)
224 {
225 	//check Ctrl Modifier
226 	if (mouseevent.HasCtrlModifier())
227 	{
228 		var raycastResult = renderer.Raycast(mouseevent.x, mouseevent.y);
229 	    if (raycastResult.entity !== null)
230 	    {
231 	    	me.Exec(1, "GoTo", [raycastResult.pos.x.toString(),raycastResult.pos.y.toString(),raycastResult.pos.z.toString(), "true"]);
232 	    	//me.Exec(1, "LookAt", raycastResult.pos.x.toString(),raycastResult.pos.y.toString(),raycastResult.pos.z.toString());
233 	    }
234 	}

237 realXtend.EntityController.SimpleAvatar.prototype.ClientKeyMove = function(param)
238 {
239 	if (param == "forward")
240 		this.motion.x = 1;
241 	else if (param == "back")
242 		this.motion.x = -1;
243 	else if (param == "left")
244 		this.motion.y = 1;
245 	else if (param == "right")
246 		this.motion.y = -1;
247 	else if (param == "up") 
248 		this.motion.z = 1;
249 	else if (param == "down")
250 		this.motion.z = -1;
251 	
252 	var move = this.motion.normalize();
253 	me.Exec(1, "Move", move.x, move.y, move.z);
254 };

265 realXtend.EntityController.SimpleAvatar.prototype.ClientKeyStop = function(param)
266 {	
267 	if ((param == "forward") && (this.motion.x > 0))
268 		this.motion.x = 0;
269 	else if ((param == "back") && (this.motion.x < 0))
270 		this.motion.x = 0;
271 	else if ((param == "left") && (this.motion.y > 0))
272 		this.motion.y = 0;
273 	else if ((param == "right") && (this.motion.y < 0))
274 		this.motion.y = 0;
275 	else if ((param == "up") && (this.motion.z > 0))
276 		this.motion.z = 0;
277 	else if ((param == "down") && (this.motion.z < 0))
278 		this.motion.z = 0;
279 	
280 	var move = this.motion.normalize();
281 	me.Exec(1, "Move", move.x, move.y, move.z);	
282 };

293 realXtend.EntityController.SimpleAvatar.prototype.GestureStarted = function(gestureEvent)
294 {
295 	if (this.client_avatar_camera_sync)
296 		if (!this.client_avatar_camera_sync.IsCameraActive())
297 			return;
298 	 
299     if (gestureEvent.GestureType() == Qt.PanGesture)
300     {
301         listenGesture = true;
302         var x = new Number(gestureEvent.Gesture().offset.toPoint().x()); 
303         // move on server z 
304         me.Exec(1, "Rotate", 0, 0, x.toString());
305         gestureEvent.Accept();
306     }
307     else if (gestureEvent.GestureType() == Qt.PinchGesture)
308         gestureEvent.Accept();
309 };

311 realXtend.EntityController.SimpleAvatar.prototype.GestureUpdated = function(gestureEvent)
312 {
313 	if (this.client_avatar_camera_sync)
314 		if (!this.client_avatar_camera_sync.IsCameraActive())
315 			return;
316 
317     if (gestureEvent.GestureType() == Qt.PanGesture && listenGesture == true)
318     {
319         // Rotate avatar with X pan gesture
320         delta = gestureEvent.Gesture().delta.toPoint();
321         var x = new Number(delta.x());
322         me.Exec(1, "Rotate", 0, 0, x.toString());
323 
324         // Start walking or stop if total Y len of pan gesture is 100
325         var walking = false;
326         if (me.animationcontroller.animationState == walkAnimName)
327             walking = true;
328         var totalOffset = gestureEvent.Gesture().offset.toPoint();
329         if (totalOffset.y() < -100)
330         {
331             if (walking) {
332                 me.Exec(2, "Move", 0,0,0);
333                 me.Exec(2, "Stop", "back");
334             } else
335                 me.Exec(2, "Move", "forward");
336             listenGesture = false;
337         }
338         else if (totalOffset.y() > 100)
339         {
340             if (walking) {
341                 me.Exec(2, "Stop", "forward");
342                 me.Exec(2, "Stop", "back");
343             } else
344                 me.Exec(2, "Move", "back");
345             listenGesture = false;
346         }
347         gestureEvent.Accept();
348     }
349     else if (gestureEvent.GestureType() == Qt.PinchGesture)
350     {
351         var scaleChange = gestureEvent.Gesture().scaleFactor - gestureEvent.Gesture().lastScaleFactor;
352         if (scaleChange > 0.1 || scaleChange < -0.1)
353             ClientHandleMouseScroll(scaleChange * 100);
354         gestureEvent.Accept();
355     }
356 };

358 realXtend.EntityController.SimpleAvatar.prototype.CreateFish = function() {
359     // Note: attaching meshes to bone of another mesh is strictly client-only! It does not replicate.
360     // Therefore this needs to be run locally on every client
361     var avatarmesh = me.GetComponentRaw("EC_Mesh", "");
362     // Do not act until the actual avatar has been created
363     if ((avatarmesh) && (avatarmesh.HasMesh())) {
364         // Create a local mesh component into the same entity
365         var fishmesh = me.GetOrCreateComponentRaw("EC_Mesh", "fish", 2, false);
366         var r = fishmesh.meshRef;
367         if (r.ref != "local://fish.mesh") {
368             r.ref = "local://fish.mesh";
369             fishmesh.meshRef = r;
370         }
371 
372         // Then we must wait until the fish mesh component has actually loaded the mesh asset
373         if (fishmesh.HasMesh()) {
374             fishmesh.AttachMeshToBone(avatarmesh, "Bip01_Head");
375             this.fish_created = true;
376             var t = fishmesh.nodeTransformation;
377             var scaleVec = new Vector3df();
378             scaleVec.x = 0.1;
379             scaleVec.y = 0.1;
380             scaleVec.z = 0.1;
381             t.scale = scaleVec;
382             fishmesh.nodeTransformation = t;
383         }
384     }
385 };
