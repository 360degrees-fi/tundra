 34 /**
 35  * Creates a PlaceableController object.
 36  * 
 37  * @class Controller of Placeable
 38  * @augments realXtend.EntityController.EntityController
 39  */
 40 realXtend.EntityController.PlaceableController = function ()
 41 {
 42 	this.velocity = new Vector3df();
 43 	//Initialize
 44 	var isserver = server.IsRunning();
 45 	if (isserver) {
 46 	    this.ServerInitialize();
 47 	} 
 48 	else {
 49 		this.ClientInitialize(); 
 50 	}
 51 };

 67 //COMMON METHODS
 68 /**
 69  * Common update method
 70  * @param {Number} frametime
 71  */
 72 realXtend.EntityController.PlaceableController.prototype.CommonUpdate = function(frametime)
 73 {
 74 	if (me.placeable)
 75 	{
 76 		this.SyncObjects(frametime, [this.velocity]);
 77 	}
 78 	
 79 	if (this.getRotateAutoStop() == true)
 80 	{
 81 		var tmpVec = new Vector3df();
 82 		tmpVec.x = 0;
 83 		tmpVec.y = 0;
 84 		tmpVec.z = 0;
 85 		this.setRotate(tmpVec);
 86 		this.setRotateAutoStop(false);
 87 	}
 88 };

 94 realXtend.EntityController.PlaceableController.prototype.ClientInitialize = function()
 95 {
 96 	if(me.IsLocal())
 97 		this.ServerInitialize();	
 98 	else
 99 		frame.Updated.connect(this, this.CommonUpdate);	
100 	
101 	// Connect actions
102 	/**
103 	 * [Action] Move client/server action
104 	 * @name realXtend.EntityController.PlaceableController#Move
105 	 * @event
106 	 * @param {Number} x
107 	 * @param {Number} y
108 	 * @param {Number} z
109 	 */
110 	me.Action("Move").Triggered.connect(this, this.Move);
111 	/**
112 	 * [Action] Rotate client/server action
113 	 * @name realXtend.EntityController.PlaceableController#Rotate
114 	 * @event
115 	 */
116 	me.Action("Rotate").Triggered.connect(this, this.Rotate);
117 
118 };

161 /**
162  * Placeable update method
163  * @param {Number} frametime
164  */
165 realXtend.EntityController.PlaceableController.prototype.ServerUpdate = function(frametime)
166 {   
167 	//move placeable
168 	var moveVec = new Vector3df();
169 	if ((this.getMove().x != 0) || (this.getMove().y != 0) || (this.getMove().z != 0))
170 	{
171 		this.ResetAutoUpdatedVariables();
172 		moveVec = this.getMove();
173 		moveVec = me.placeable.GetRelativeVector(moveVec);
174 	}
175 	else if (this.getGoTo() != 0)
176 	{
177 		var placeable_pos = me.placeable.transform.pos;
178 		var dest = new Vector3df();
179 		dest.x = this.getGoTo().x;
180 		dest.y = this.getGoTo().y;
181 		dest.z = this.getGoTo().z;
182 			
183 		if ((Math.abs(placeable_pos.x - dest.x) < this.goToThreshold) && (Math.abs(placeable_pos.y - dest.y) < this.goToThreshold) && (Math.abs(placeable_pos.z - dest.z) < this.goToThreshold))
184 		{
185 			this.setGoTo(0);
186 		}
187 		else
188 		{
189 			//set move vector 			
190 			moveVec.x = dest.x - placeable_pos.x;
191 			moveVec.y = dest.y - placeable_pos.y;
192 			moveVec.z = dest.z - placeable_pos.z;
193 			
194 			moveVec = moveVec.normalize();
195 			//print("goTo moveVec = x:" + moveVec.x + " y:" + moveVec.y + " z:" + moveVec.z);
196 			
197 			//set lookat if goTolooking is true
198 			if (this.getGoToLooking() == true)
199 				this.setLookAt(this.getGoTo());
200 		}
201 	}
202 	//if there is movement
203 	if ((moveVec.x != 0) || (moveVec.y != 0) || (moveVec.z != 0))
204 		this.Move(moveVec.x, moveVec.y, moveVec.z, frametime);
205 		
206 	//rotate placeable
207 	var rotateVec = new Vector3df();
208 	if ((this.getRotate().x != 0) || (this.getRotate().y != 0) || (this.getRotate().z != 0))
209 	{
210 		this.ResetAutoUpdatedVariables();
211 		rotateVec = this.getRotate();
212 	}
213 	else if ((this.getLookAt() != 0) || (this.getRotateTo() != 0))
214 	{
215 		var lookAtRot = 0;
216 		if (this.getLookAt() != 0) //set lookAtRot for lookat variable
217 		{
218 			var lookAt = this.getLookAt();
219 			var rotationMaskX = this.syncProperties.GetAttribute("rotationMaskX");
220 			var rotationMaskY = this.syncProperties.GetAttribute("rotationMaskY");
221 					
222 			var diff = new Vector3df();
223 			var current_pos = me.placeable.transform.pos;
224 			diff.x = lookAt.x - current_pos.x;
225 			diff.y = lookAt.y - current_pos.y;
226 			diff.z = lookAt.z - current_pos.z;		
227 			if ((rotationMaskX==0) && (rotationMaskY==0))
228 				diff.z = 0;
229 			
230 			var lookAtDirectionX = this.syncProperties.GetAttribute("lookAtDirectionX");
231 			var lookAtDirectionY = this.syncProperties.GetAttribute("lookAtDirectionY");
232 			var lookAtDirectionZ = this.syncProperties.GetAttribute("lookAtDirectionZ");
233 			var mov_dir = new Vector3df();
234 
235 			mov_dir.x = lookAtDirectionX;
236 			mov_dir.y = lookAtDirectionY;
237 			mov_dir.z = lookAtDirectionZ;
238 			
239 			mov_dir = me.placeable.GetRelativeVector(mov_dir);
240 			
241 			lookAtRot = me.placeable.GetRotationFromTo(mov_dir,diff);			
242 		}
243 		else if (this.getRotateTo() != 0)  //set lookAtRot for rotateTo variable
244 		{
245 			lookAtRot = new Vector3df();			
246 			lookAtRot.x	=  this.getRotateTo().x - me.placeable.transform.rot.x;
247 			lookAtRot.y =  this.getRotateTo().y - me.placeable.transform.rot.y;
248 			lookAtRot.z =  this.getRotateTo().z - me.placeable.transform.rot.z;
249 			lookAtRot = this.NormalizeRotation(lookAtRot);			
250 		}		
251 		
252 		if (lookAtRot != 0)
253 		{
254 			//print(lookAtRot.x + " "+ lookAtRot.y +" "+lookAtRot.z);
255 			//rotate with default velocity (set 1 or -1 rotateVec)
256 			if (lookAtRot.z > this.lookAtThreshold)
257 				rotateVec.z = 1;
258 			else if (lookAtRot.z < -this.lookAtThreshold)
259 				rotateVec.z = -1;
260 			if (lookAtRot.y > this.lookAtThreshold)
261 				rotateVec.y = 1;
262 			else if (lookAtRot.y < -this.lookAtThreshold)
263 				rotateVec.y = -1;			
264 			if (lookAtRot.x > this.lookAtThreshold)
265 				rotateVec.x = 1;
266 			else if (lookAtRot.x < -this.lookAtThreshold)
267 				rotateVec.x = -1;
268 
269 			
270 			if ((rotateVec.x == 0) && (rotateVec.y == 0) && (rotateVec.z == 0))
271 			{	
272 				this.setLookAt(0);	
273 				this.setRotateTo(0);
274 			}		
275 		}
276 			
277 	}
278 	//if there is rotation
279 	if ((rotateVec.x != 0) || (rotateVec.y != 0) || (rotateVec.z != 0))
280 		this.Rotate(rotateVec.x, rotateVec.y, rotateVec.z, frametime);   
281 		
282 	this.CommonUpdate(frametime);
283 };

285 /**
286  * Rotate action
287  * @param {Number} x
288  * @param {Number} y
289  * @param {Number} z
290  */
291 realXtend.EntityController.PlaceableController.prototype.Rotate = function(x, y, z) 
292 {
293 	if ((x != 0) || (y != 0) || (z != 0))
294 	{
295 		var velocity = this.syncProperties.GetAttribute("rotationVelocity");
296 		if (!velocity)
297 			return;
298 		
299 		var rotationMaskX = this.syncProperties.GetAttribute("rotationMaskX");
300 		var rotationMaskY = this.syncProperties.GetAttribute("rotationMaskY");
301 		var rotationMaskZ = this.syncProperties.GetAttribute("rotationMaskZ");
302 		
303 		if (isNaN(rotationMaskX) || isNaN(rotationMaskY) || isNaN(rotationMaskZ))
304 			return;
305 		
306         var rotateVec = new Vector3df();
307         rotateVec.x = x *rotationMaskX;
308         rotateVec.y = y *rotationMaskY;
309         rotateVec.z = z *rotationMaskZ;
310 		
311         var v = new Vector3df();
312         v.x = velocity * rotateVec.x/180 * Math.PI;
313         v.y = velocity * rotateVec.y/180 * Math.PI;
314         v.z = velocity * rotateVec.z/180 * Math.PI;
315 		
316 		var placeable = me.placeable;
317         var transform = placeable.transform;
318 
319 		transform.rot.x=transform.rot.x+v.x;
320 		transform.rot.y=transform.rot.y+v.y;
321 		transform.rot.z=transform.rot.z+v.z;
322 		
323 	    placeable.transform = transform;	
324     }
325 };

327 /**
328  * Move action
329  * @param {Number} x
330  * @param {Number} y
331  * @param {Number} z
332  */
333 realXtend.EntityController.PlaceableController.prototype.Move = function(x, y, z, frametime)
334 {
335 	var moveVec = new Vector3df();
336 	moveVec.x = x;
337 	moveVec.y = y;
338 	moveVec.z = z;
339 	
340 	var velocity = this.syncProperties.GetAttribute("movementVelocity");
341 	if (!velocity)
342 		return;		
343 
344 	// translation
345 	var translation = new Vector3df();
346 
347 	translation.x = velocity*moveVec.x*frametime;
348 	translation.y = velocity*moveVec.y*frametime;
349 	translation.z = velocity*moveVec.z*frametime;
350 	
351 	me.placeable.Translate(translation);
352 	
353 	this.velocity.x=translation.x/frametime;
354 	this.velocity.y=translation.y/frametime;
355 	this.velocity.z=translation.z/frametime;
356 };
