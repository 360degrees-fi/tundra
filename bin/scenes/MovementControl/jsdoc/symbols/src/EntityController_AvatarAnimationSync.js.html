 12 realXtend.EntityController.AvatarAnimationSync = function()
 13 {
 14 	//Animation detection
 15 	this.standAnimName = "Stand";
 16 	this.walkAnimName = "Walk";
 17 	this.runAnimName = "Run";
 18 	this.flyAnimName = "Fly";
 19 	this.hoverAnimName = "Hover";
 20 	this.sitAnimName = "SitOnGround";
 21 	this.waveAnimName = "Wave";
 22 	this.animsDetected = false;
 23 	
 24 	this.walk_anim_speed = 0.5;
 25 	this.run_anim_speed = 0.25;
 26 	
 27 	this.velocityTrigger = 0.1;
 28 
 29 	this.isserver = server.IsRunning();
 30 	if (!this.isserver) {
 31 		me.Action("Gesture").Triggered.connect(this, this.ClientHandleGesture);
 32 	}
 33 	this.CommonFindAnimations();
 34 };

 36 realXtend.EntityController.AvatarAnimationSync.prototype.Update = function(frametime, params) {
 37 	var velocity = params[0];
 38 	var flying = params[1];
 39 	var running = params[2];
 40 	var onGround = params[3];
 41 	if (this.isserver) {
 42 		this.ServerSetAnimationState(velocity, flying, running, onGround);
 43 		this.CommonUpdateAnimation(frametime, velocity);
 44 	}
 45 	else {
 46 		this.CommonUpdateAnimation(frametime, velocity);
 47 	}
 48 };

 51 realXtend.EntityController.AvatarAnimationSync.prototype.CommonFindAnimations = function () {
 52     var animcontrol = me.animationcontroller;
 53     if (animcontrol == null)
 54     	return;
 55     var availableAnimations = animcontrol.GetAvailableAnimations();
 56     if (availableAnimations.length > 0) {
 57         // Detect animation names
 58         var searchAnims = [this.standAnimName, this.walkAnimName, this.flyAnimName, this.hoverAnimName, this.sitAnimName, this.waveAnimName];
 59         for(var i=0; i<searchAnims.length; i++) {
 60             var animName = searchAnims[i];
 61             if (availableAnimations.indexOf(animName) == -1) {
 62                 // Disable this animation by setting it to a empty string
 63                 print("Could not find animation for:", animName, " - disabling animation");
 64                 searchAnims[i] = "";
 65             }
 66         }
 67 
 68         // Assign the possible empty strings for
 69         // not found anims back to the variables
 70         this.standAnimName = searchAnims[0];
 71         this.walkAnimName = searchAnims[1];
 72         this.flyAnimName = searchAnims[2];
 73         this.hoverAnimName = searchAnims[3];
 74         this.sitAnimName = searchAnims[4];
 75 
 76         this.animsDetected = true;
 77     }
 78 };

 80 realXtend.EntityController.AvatarAnimationSync.prototype.CommonUpdateAnimation = function(frametime, velocity) {
 81     if (!this.animsDetected) {
 82     	this.CommonFindAnimations();
 83         return;
 84     }
 85 
 86     var animcontroller = me.animationcontroller;
 87     if (animcontroller == null){
 88         return;
 89     }
 90 
 91     var animName = animcontroller.animationState;
 92 
 93     // Enable animation, skip with headless server
 94     if (animName != "" && !framework.IsHeadless()) {
 95         // Do custom speeds for certain anims
 96         if (animName == this.hoverAnimName) {
 97             animcontroller.SetAnimationSpeed(animName, 0.25);
 98         }
 99         if (animName == this.sitAnimName) { // Does not affect the anim speed on jack at least?!
100             animcontroller.SetAnimationSpeed(animName, 0.5);
101         }
102         if (animName == this.waveAnimName) {
103             animcontroller.SetAnimationSpeed(animName, 0.75);
104         }
105         // Enable animation
106         if (!animcontroller.IsAnimationActive(animName)) {
107             // Gestures with non exclusive
108             if (animName == this.waveAnimName) {
109                 animcontroller.EnableAnimation(animName, false, 0.25, 0.25, false);
110             // Normal anims exclude others
111             } else {
112                 animcontroller.EnableExclusiveAnimation(animName, true, 0.25, 0.25, false);
113             }
114         }
115     }
116 
117  // If walk animation is playing, adjust its speed according to the avatar rigidbody velocity
118     if (animName != ""  && animcontroller.IsAnimationActive(this.walkAnimName)) {
119         // Note: on client the rigidbody does not exist, so the velocity is only a replicated attribute
120         var walkspeed = Math.sqrt(velocity.x * velocity.x + velocity.y * velocity.y) * this.walk_anim_speed;        
121         if (velocity.x < -this.velocityTrigger)
122         	walkspeed = -walkspeed;
123         animcontroller.SetAnimationSpeed(animName, walkspeed);
124     }
125     else if (animName != ""  && animcontroller.IsAnimationActive(this.runAnimName)) {
126         // Note: on client the rigidbody does not exist, so the velocity is only a replicated attribute
127         var runspeed = Math.sqrt(velocity.x * velocity.x + velocity.y * velocity.y) * this.run_anim_speed;
128         if (velocity.x < -this.velocityTrigger)
129         	runspeed = -runspeed;
130         animcontroller.SetAnimationSpeed(animName, runspeed);
131     }
132 };

135 realXtend.EntityController.AvatarAnimationSync.prototype.ClientHandleGesture = function(gestureName) {
136     var animName = "";
137     if (gestureName == "wave") {
138         animName = this.waveAnimName;
139     }
140     if (animName == "") {
141         return;
142     }
143 
144     // Update the variable to sync to client if changed
145     var animcontroller = me.animationcontroller;
146     if (animcontroller != null) {
147         if (animcontroller.animationState != animName) {
148             animcontroller.animationState = animName;
149 	}
150     }
151 };

154 realXtend.EntityController.AvatarAnimationSync.prototype.ServerSetAnimationState = function(velocity, flying, running, onGround) {
155 
156     var animName = this.standAnimName;
157     // Not flying: Stand, Walk, Run, Crouch, Jump or Fall
158     if (!flying)
159     {
160     	if (velocity.z > this.velocityTrigger) { //Jump
161     		animName = this.hoverAnimName;
162     	}
163     	else if (velocity.z < - this.velocityTrigger){ //Crouch or Fall
164     		if (onGround) {
165     			animName = this.sitAnimName;
166     		}
167     		else {
168     			animName = this.hoverAnimName;
169     		}
170     	}
171     	else { //Stand or Walk
172     		if ((Math.abs(velocity.x) > this.velocityTrigger) || (Math.abs(velocity.y) > this.velocityTrigger)) {
173         		if (running)
174         			animName = this.runAnimName;
175         		else
176         			animName = this.walkAnimName;
177         	}
178     	}
179     }
180     else { // Flying: Fly if moving in x-axis, otherwise hover
181     	animName = this.flyAnimName;
182         if (Math.abs(velocity.x) < this.velocityTrigger)
183         	animName = this.hoverAnimName;
184     }
185 
186     if (animName == "") {
187         return;
188     }
189 
190     // Update the variable to sync to client if changed
191     var animcontroller = me.animationcontroller;
192     if (animcontroller != null) {
193         if (animcontroller.animationState != animName) {
194             animcontroller.animationState = animName;
195 	}
196     }
197 };
